96 unsigned int loadCubemap(std::vector<std::string> faces) {
97 unsigned int textureID;
98 glGenTextures(1, &textureID);
99 glBindTexture(GL_TEXTURE_CUBE_MAP, textureID);
100
101 // IMPORTANTE: Desativar flip vertical para cubemaps
102 // (o flip é usado para texturas normais, mas não para skybox)
103 stbi_set_flip_vertically_on_load(false);
104
105 int width, height, nrChannels;
106 for (unsigned int i = 0; i < faces.size(); i++)
107 {
108 unsigned char *data = stbi_load(faces[i].c_str(), &width, &height, &nrChannels, 0);
109 if (data)
110 {
111 GLenum format = (nrChannels == 3) ? GL_RGB : GL_RGBA;
112 glTexImage2D(GL_TEXTURE_CUBE_MAP_POSITIVE_X + i, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, data);
113 stbi_image_free(data);
114 std::cout << "Skybox face " << i << " carregada: " << faces[i] << std::endl;
115 }
116 else
117 {
118 std::cout << "Cubemap load failed at: " << faces[i] << std::endl;
119 stbi_image_free(data);
120 }
121 }
122
123 // Configurar filtros de textura para melhor qualidade
124 glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
125 glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
126 glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
127 glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
128 glTexParameteri(GL_TEXTURE_CUBE_MAP, GL_TEXTURE_WRAP_R, GL_CLAMP_TO_EDGE);
129
130 // Ativar seamless cubemap para eliminar costuras entre faces (OpenGL 3.2+)
131 glEnable(GL_TEXTURE_CUBE_MAP_SEAMLESS);
132
133 // Reativar flip vertical para outras texturas que possam ser carregadas depois
134 stbi_set_flip_vertically_on_load(true);
135
136 return textureID;
137 }
