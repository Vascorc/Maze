50 shader.use();

56 float scrAspect = scrWidth / scrHeight;
57 glm::vec2 scale(1.0f);
58
59 // Fit logic: Manter aspect ratio da imagem dentro do ecrã
60 if (scrAspect > imgAspect) {
61 // Ecrã mais largo que a imagem (barras laterais)
62 scale.x = imgAspect / scrAspect;
63 } else {
64 // Ecrã mais alto que a imagem (barras em cima/baixo)
65 scale.y = scrAspect / imgAspect;
66 }
67 shader.setVec2("scale", scale);
